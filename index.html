WEATHER STATION SYSTEM

THIS DOCUMENTATION DETAILS THE WEATHER STATION SYSTEM PROJECT, WHICH UTILIZES THE DHT22 SENSOR TO ACCURATELY MEASURE AND LOG ENVIRONMENTAL TEMPERATURE AND HUMIDITY DATA.

SOURCE CODE

THE ARDUINO SKETCH USES THE DHT LIBRARY TO READ DATA AND SENDS THE HUMIDITY, CELSIUS, AND FAHRENHEIT READINGS TO THE SERIAL MONITOR EVERY TWO SECONDS.

COPIED!

#include "DHT.h" // Include the DHT Sensor library

#define DHTPIN 2       // Digital pin connected to the DHT sensor (e.g., Arduino D2)
#define DHTTYPE DHT22  // Specify the sensor type: DHT 22 (AM2302)

// Initialize DHT sensor instance
DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(9600); // Initialize serial communication at 9600 bits per second
  Serial.println("DHT22 WEATHER STATION INITIATED.");
  dht.begin(); // Start the DHT sensor
}

void loop() {
  // DHT sensor data acquisition is slow, so we wait 2 seconds between readings
  delay(2000); 

  // Read humidity and temperature (in Celsius) from the sensor
  float h = dht.readHumidity();
  float t = dht.readTemperature();

  // --- ERROR CHECKING ---
  // Check if any reads failed (returns 'NaN' - Not a Number) and handle the error
  if (isnan(h) || isnan(t)) {
    Serial.println("!!! FAILED TO READ FROM DHT SENSOR. CHECK WIRING AND POWER. !!!");
    return; // Exit the loop iteration early
  }

  // --- DATA CONVERSION ---
  // Convert Celsius to Fahrenheit using the formula: F = C * 9/5 + 32
  float f = t * 9.0 / 5.0 + 32.0;

  // --- SERIAL OUTPUT ---
  Serial.print("HUMIDITY: ");
  Serial.print(h);
  Serial.print(" %\t");
  Serial.print("TEMPERATURE: ");
  Serial.print(t);
  Serial.print(" *C / ");
  Serial.print(f);
  Serial.println(" *F");
}
